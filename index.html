const closeMenu = (event) => {
  if (event.target.classList.contains('nav-link')) {
    hamburger.classList.remove('actv');
    menuMob.classList.remove('actv');
  }
};

navLinks.forEach((el) => el.addEventListener('click', closeMenu));
